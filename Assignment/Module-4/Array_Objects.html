Q.40 Write a JavaScript Program to display the current day and time in the following format. Sample Output: Today is Friday. Current Time is 12 PM: 12 : 22 2 ?

Ans.
                    let date = new Date();
                    let day = date.getDay();
                    let daylist = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
                    let hour = date.getHours();
                    let minute = date.getMinutes();
                    let second = date.getSeconds();
                    let prepand = (hour >= 12) ? " PM " : " AM ";
                    hour = (hour >= 12) ? hour - 12 : hour;

                    if (hour === 0 && prepand === ' PM ') {
                        if (minute === 0 && second === 0) {
                            hour = 12;
                            prepand = ' Noon';
                        } else {
                            hour = 12;
                            prepand = ' PM';
                        }
                    }

                    if (hour === 0 && prepand === ' AM ') {
                        if (minute === 0 && second === 0) {
                            hour = 12;
                            prepand = ' Midnight';
                        } else {
                            hour = 12;
                            prepand = ' AM';
                        }
                    }
                    
                    console.log(`Today is ${daylist[day]}. Current Time is ${hour}${prepand}:${minute} ${second} ` );
